	\begin{ttkp}
		Gọi $x$ là số cam, $y$ là số quýt (với $x$, $y$ nguyên dương và nhỏ hơn 17).
		\begin{itemize}
			\item \textbf{HĐ1:} Câu ``Quýt, cam mười bảy quả tươi'' có nghĩa là tổng số cam và số quýt là 17. Hãy viết hệ thức với hai biến $x$ và $y$ biểu thị giả thiết này.
			\item \textbf{HĐ2:} Tương tự, hãy viết hệ thức với hai biến $x$ và $y$ biểu thị giả thiết cho bởi các câu thơ thứ ba, thứ tư và thứ năm.
		\end{itemize}
		%%%
		\doicot
		%%%
		\begin{itemize}
			\item \textbf{HĐ1:} $x+y=17$
			\item \textbf{HĐ2:} Ta có $3y$; $10x$ và hệ thức liên hệ là $$10x+3y=100.$$
		\end{itemize}
	\end{ttkp}
	
	\begin{vidu}
		\begin{listEX}
			\item Trong các hệ thức $4x + 3y = 5$; $0x + y = -1$; $0x + 0y = 3$, hệ thức nào là phương trình bậc nhất hai ẩn? Hệ thức nào không là phương trình bậc nhất hai ẩn?
			\item Trong các cặp số $(2; -1)$ và $(1; 0)$, cặp số nào là nghiệm của phương trình $4x + 3y = 5$?
		\end{listEX}
		%%%
		\doicot
		%%%
		\begin{listEX}
			\item Cả ba hệ thức đều có dạng $ax + by = c$. Nhưng chỉ có hai hệ thức $4x + 3y = 5$ và $0x + y = -1$ thỏa mãn điều kiện $a \neq 0$ hoặc $b \neq 0$ nên là phương trình bậc nhất hai ẩn.
			
			Hệ thức $0x + 0y = 3$ có $a = b = 0$, không thỏa mãn điều kiện trên nên hệ thức đó không phải là phương trình bậc nhất hai ẩn.
			
			\item Cặp số $(2; -1)$ là một nghiệm của phương trình $4x + 3y = 5$, vì
			\[ 4 \cdot 2 + 3 \cdot (-1) = 5. \]
			Cặp số $(1; 0)$ \textit{không} là nghiệm của phương trình $4x + 3y = 5$, vì
			\[ 4 \cdot 1 + 3 \cdot 0 = 4 \neq 5. \]
		\end{listEX}
	\end{vidu}

	\begin{vidu}
		Giả sử $(x; y)$ là nghiệm của phương trình bậc nhất hai ẩn $x + 2y = 5$.
		
		\begin{listEX}
			\item Hoàn thành bảng sau đây:
			
			\renewcommand{\arraystretch}{1.5} % Giãn dòng cho bảng
			\begin{center}
				\begin{tabular}{|c|c|c|c|c|c|}
					\hline
					$x$ & $-2$ & $-1$ & $0$ & ? & ? \\
					\hline
					$y$ & ? & ? & ? & $1$ & $2$ \\
					\hline
				\end{tabular}
			\end{center}
			
			Từ đó suy ra 5 nghiệm của phương trình đã cho.
			
			\item Biểu diễn $y$ theo $x$. Từ đó cho biết phương trình đã cho có bao nhiêu nghiệm?
		\end{listEX}
		%%%
		\doicot
		%%%		
		\begin{listEX}
			\item Ta có:
			
			\renewcommand{\arraystretch}{2} % Giãn dòng cao hơn để chứa phân số
			\begin{center}
				\begin{tabular}{|c|c|c|c|c|c|}
					\hline
					$x$ & $-2$ & $-1$ & $0$ & $3$ & $1$ \\
					\hline
					$y$ & $\dfrac{7}{2}$ & $3$ & $\dfrac{5}{2}$ & $1$ & $2$ \\
					\hline
				\end{tabular}
			\end{center}
			
			Vậy 5 nghiệm của phương trình đã cho là: $\left(-2; \dfrac{7}{2}\right)$ $(-1; 3)$, $\left(0; \dfrac{5}{2}\right)$, $(3; 1)$, $(1; 2)$.
			
			\item Ta có $y = \dfrac{5 - x}{2}$. Với mỗi giá trị $x$ tùy ý cho trước, ta luôn tìm được một giá trị $y$ tương ứng. Do đó phương trình đã cho có vô số nghiệm.
		\end{listEX}
	\end{vidu}

	\begin{paracol}{2}

	\begin{dhnh}
		Trong HĐ1 và HĐ2, bài toán mở đầu dẫn đến hai phương trình bậc nhất hai ẩn là $x + y = 17$ và $10x + 3y = 100$. Để giải bài toán, ta cần tìm các giá trị của $x$ và $y$ đồng thời thoả mãn cả hai phương trình này. Khi đó ta có hệ phương trình $$\heva{&x+y=17\\&10x+3y=100.}$$
	\end{dhnh}
	
	\paragraph{Tổ chức thực hiện}
	
	\begin{tcth}
		\buocmot
		\item GV cho HS tự đọc phần Đọc hiểu - Nghe hiểu.
		\buochai
		\item HS đọc thông tin.
		\buocba
		\item HS thảo luận các chi tiết chưa hiểu.
		\buocbon
		\item GV viết bảng hoặc trình chiếu nội dung trong Khung kiến thức:
		\begin{kttt}
			\begin{itemize}[leftmargin=13pt]
				\item Một cặp gồm hai phương trình bậc nhất hai ẩn $ax + by = c$ và $a'x + b'y = c'$ được gọi là một \textit{hệ hai phương trình bậc nhất hai ẩn}. Ta thường viết hệ phương trình đó dưới dạng:
				$$\heva{&ax + by = c\\&a'x + b'y = c'.}\quad (*)$$
				\item Mỗi cặp số $(x_0; y_0)$ được gọi là một \textit{nghiệm} của hệ $(*)$ nếu nó đồng thời là nghiệm của cả hai phương trình của hệ $(*)$.
			\end{itemize}
		\end{kttt}
		\item GV nhấn mạnh các ý: 
		\begin{itemize}
			\item Cách viết hệ phương trình, trong đó thứ tự các phương trình trong hệ là không quan trọng. 
			\item Nghiệm của hệ là nghiệm chung của các phương trình trong hệ. 
			\item Cách viết nghiệm của một hệ phương trình, trong đó giá trị của $x$ luôn đứng trước giá trị của $y$. 
		\end{itemize}
		\item HS ghi nội dung bài học vào vở. 
	\end{tcth}

	\begin{vidu}
		Trong các hệ phương trình sau, hệ nào \textit{không} phải là hệ hai phương trình bậc nhất hai ẩn? Vì sao?
		\begin{listEX}
			\item $\heva{&2x = -6 \\ &5x + 4y = 1;}$
			\item $\heva{&x + 2y = -3 \\ &0x + 0y = 1;}$
			\item $\heva{&23x - y = 1 \\ &x + y = 3.}$
		\end{listEX}
		%%%
		\doicot
		%%%
		Hệ phương trình b) không phải là hệ hai phương trình bậc nhất hai ẩn, vì phương trình thứ hai của hệ là $0x + 0y = 1$ không phải là phương trình bậc nhất hai ẩn.
	\end{vidu}
	
	\begin{vidu}
		Giải thích tại sao cặp số $(1; 2)$ là một nghiệm của hệ phương trình $\heva{ &2x - y = 0 \\ &x + y = 3.}$
		%%%
		\doicot
		%%%
		Ta thấy khi $x = 1$ và $y = 2$ thì:
		\begin{itemize}
			\item $2x - y = 2 \cdot 1 - 2 = 0$ nên $(1; 2)$ là nghiệm của phương trình thứ nhất;
			\item $x + y = 1 + 2 = 3$ nên $(1; 2)$ là nghiệm của phương trình thứ hai.
		\end{itemize}
		Vậy $(1; 2)$ là nghiệm chung của hai phương trình, nghĩa là $(1; 2)$ là một nghiệm của hệ phương trình đã cho.
	\end{vidu}
	
	\begin{luyentap}
		Trong hai cặp số $(0;-2)$ và $(2;-1)$, cặp số nào là nghiệm của hệ phương trình
		$$\heva{&x-2y=4\\&4x+3y=5}?$$
		%%%
		\doicot
		%%%
		Xét cặp số $(2;-1)$, tức là $x = 2$ và $y = 2$ thì:
		\begin{itemize}[leftmargin=10pt,label*=$\bullet$]
			\item $x-2y=2-2\cdot(-1)=4$ nên $(2; -1)$ là nghiệm của phương trình thứ nhất;
			\item $4x+3y=3\cdot2+3\cdot(-1)=5$ nên $(2; -1)$ là nghiệm của phương trình thứ hai.
		\end{itemize}
		Vậy $(2; -1)$ là nghiệm chung của hai phương trình, nghĩa là $(2; -1)$ là một nghiệm của hệ phương trình đã cho.
	\end{luyentap}

	\subsection{Phụ lục}
	
	\begin{ex}
		Phương trình nào sau đây KHÔNG là phương trình bậc nhất hai ẩn?
		\choice
		{$x - 2y = 5$}
		{$0x + 0y = -3$}
		{$6x + 0y = 1$}
		{$0x - 4y = 3$}
	\end{ex}
	
	\begin{ex}
		Phương trình $3x + y = -2$ có nghiệm là cặp số nào sau đây?
		\choice
		{$(1; -5)$}
		{$(-1; -1)$}
		{$(0; 2)$}
		{$(2; 4)$}
	\end{ex}
	
	\begin{ex}
		Phương trình nào sau đây nhận cặp số $(-2; 3)$ làm nghiệm?
		\choice
		{$2x + 3y = -5$}
		{$2x - 3y = 5$}
		{$-2x + 3y = 5$}
		{$2x + 3y = 5$}
	\end{ex}
	
	\begin{ex}
		Nghiệm tổng quát của phương trình bậc nhất hai ẩn $5x - 2y = 4$ là
		\choice
		{$\left(x; \dfrac{5}{2}x + 2\right)$ với $x \in \mathbb{R}$}
		{$\left(x; \dfrac{-5}{2}x + 2\right)$ với $x \in \mathbb{R}$}
		{$\left(x; \dfrac{5}{2}x - 2\right)$ với $x \in \mathbb{R}$}
		{$\left(x; \dfrac{-5}{2}x - 2\right)$ với $x \in \mathbb{R}$}
	\end{ex}
	
	\begin{ex}
		Cặp số nào sau đây là nghiệm của hệ phương trình $\heva{&3x+y=3\\&2x+5y=-11}?$
		\choice
		{$(-2; -3)$}
		{$(-2; 3)$}
		{$(2; -3)$}
		{$(2; 3)$}
	\end{ex}
	
	\begin{ex}
		Cặp số $\left(\dfrac{1}{2}; 4\right)$ là nghiệm của hệ phương trình nào sau đây?
		\choice
		{$\heva{&2x-y=-3\\&6x-3y=15}$}
		{$\heva{&2x-y=-3\\&6x+3y=15}$}
		{$\heva{&2x+y=-3\\&6x+3y=15}$}
		{$\heva{&-2x+y=-3\\&6x+3y=15}$}
	\end{ex}
