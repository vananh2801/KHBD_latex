	\begin{ttkp}
		Gọi $x$ là số cam, $y$ là số quýt (với $x$, $y$ nguyên dương và nhỏ hơn 17).
		\begin{itemize}
			\item \textbf{HĐ1:} Câu ``Quýt, cam mười bảy quả tươi'' có nghĩa là tổng số cam và số quýt là 17. Hãy viết hệ thức với hai biến $x$ và $y$ biểu thị giả thiết này.
			\item \textbf{HĐ2:} Tương tự, hãy viết hệ thức với hai biến $x$ và $y$ biểu thị giả thiết cho bởi các câu thơ thứ ba, thứ tư và thứ năm.
		\end{itemize}
		%%%
		\doicot
		%%%
		\begin{itemize}
			\item \textbf{HĐ1:} $x+y=17$
			\item \textbf{HĐ2:} Ta có $3y$; $10x$ và hệ thức liên hệ là $$10x+3y=100.$$
		\end{itemize}
	\end{ttkp}
	
	\begin{vidu}
		\begin{listEX}
			\item Trong các hệ thức $4x + 3y = 5$; $0x + y = -1$; $0x + 0y = 3$, hệ thức nào là phương trình bậc nhất hai ẩn? Hệ thức nào không là phương trình bậc nhất hai ẩn?
			\item Trong các cặp số $(2; -1)$ và $(1; 0)$, cặp số nào là nghiệm của phương trình $4x + 3y = 5$?
		\end{listEX}
		%%%
		\doicot
		%%%
		\begin{listEX}
			\item Cả ba hệ thức đều có dạng $ax + by = c$. Nhưng chỉ có hai hệ thức $4x + 3y = 5$ và $0x + y = -1$ thỏa mãn điều kiện $a \neq 0$ hoặc $b \neq 0$ nên là phương trình bậc nhất hai ẩn.
			
			Hệ thức $0x + 0y = 3$ có $a = b = 0$, không thỏa mãn điều kiện trên nên hệ thức đó không phải là phương trình bậc nhất hai ẩn.
			
			\item Cặp số $(2; -1)$ là một nghiệm của phương trình $4x + 3y = 5$, vì
			\[ 4 \cdot 2 + 3 \cdot (-1) = 5. \]
			Cặp số $(1; 0)$ \textit{không} là nghiệm của phương trình $4x + 3y = 5$, vì
			\[ 4 \cdot 1 + 3 \cdot 0 = 4 \neq 5. \]
		\end{listEX}
	\end{vidu}

	\begin{vidu}
		Giả sử $(x; y)$ là nghiệm của phương trình bậc nhất hai ẩn $x + 2y = 5$.
		
		\begin{listEX}
			\item Hoàn thành bảng sau đây:
			
			\renewcommand{\arraystretch}{1.5} % Giãn dòng cho bảng
			\begin{center}
				\begin{tabular}{|c|c|c|c|c|c|}
					\hline
					$x$ & $-2$ & $-1$ & $0$ & ? & ? \\
					\hline
					$y$ & ? & ? & ? & $1$ & $2$ \\
					\hline
				\end{tabular}
			\end{center}
			
			Từ đó suy ra 5 nghiệm của phương trình đã cho.
			
			\item Biểu diễn $y$ theo $x$. Từ đó cho biết phương trình đã cho có bao nhiêu nghiệm?
		\end{listEX}
		%%%
		\doicot
		%%%		
		\begin{listEX}
			\item Ta có:
			
			\renewcommand{\arraystretch}{2} % Giãn dòng cao hơn để chứa phân số
			\begin{center}
				\begin{tabular}{|c|c|c|c|c|c|}
					\hline
					$x$ & $-2$ & $-1$ & $0$ & $3$ & $1$ \\
					\hline
					$y$ & $\dfrac{7}{2}$ & $3$ & $\dfrac{5}{2}$ & $1$ & $2$ \\
					\hline
				\end{tabular}
			\end{center}
			
			Vậy 5 nghiệm của phương trình đã cho là: $\left(-2; \dfrac{7}{2}\right)$ $(-1; 3)$, $\left(0; \dfrac{5}{2}\right)$, $(3; 1)$, $(1; 2)$.
			
			\item Ta có $y = \dfrac{5 - x}{2}$. Với mỗi giá trị $x$ tùy ý cho trước, ta luôn tìm được một giá trị $y$ tương ứng. Do đó phương trình đã cho có vô số nghiệm.
		\end{listEX}
	\end{vidu}

		\item GV tổ chức cho HS làm việc theo nhóm đôi trong 3 phút. 
		\buochai
		\item GV mời hai nhóm trình bày nội dung thảo luận của nhóm mình.
		\buocba
		\item Các nhóm khác nhận xét.
		\buocbon
		\item GV nhận xét bài làm của các nhóm và chốt lại nội dung.
	\end{tcth}
	
	\noindent\faBookmarkO~\textit{\textbf{Ví dụ 2:}}
	
	\begin{tcth}
		\buocmot
		\item GV sử dụng bảng phụ hoặc trình chiếu nội dung Ví dụ 2 trong SGK. 
		\item GV yêu cầu HS thực hiện ý a) của Ví dụ 2 trong 2 phút.
		\item GV yêu cầu HS thảo luận ý b) theo nhóm hai bạn cùng bàn.
		\buochai
		\item HS hoạt động cá nhân để hoàn thành bảng giá trị. Sau đó GV
		gọi một HS hoàn thành bảng giá trị.
		\item HS thảo luận yêu cầu của ý b)	với bạn để rút ra được kết luận phương trình bậc nhất hai ẩn luôn có vô số nghiệm.
		\buocba
		\item Các HS khác nhận xét.
		\buocbon
		\item GV nhận xét và rút ra Chú ý:
		\begin{chuy}
			Mỗi phương trình bậc nhất hai ẩn đều có vô số nghiệm. Ví dụ dưới đây trình bày cách viết các nghiệm và biểu diễn hình học tất cả các nghiệm của một phương trình bậc nhất hai ẩn.
		\end{chuy}
	\end{tcth}
	
	\noindent\faBookmarkO~\textit{\textbf{Ví dụ 3:}}
	
	\begin{tcth}
		\buocmot
		\item GV hướng dẫn HS giải câu a) của Ví dụ 3. 
		\item GV nhắc lại cách vẽ đồ thị của hàm số bậc nhất: Lấy hai điểm thuộc đồ thị (thường là giao điểm	với hai trục toạ độ), đường thẳng nối hai điểm chính là đồ thị cần vẽ.
		\item GV yêu cầu HS thực hiện ý b), c) của Ví dụ 3. 
		\buochai
		\item HS hoàn thành bài làm.
		\item GV gọi hai HS lên bảng làm bài.
		\buocba
		\item Các HS khác nhận xét.
		\buocbon
		\item GV nhận xét và rút ra Nhận xét:
		\begin{nhanxet}
			Trong mặt phẳng tọa độ, tập hợp các điểm có tọa độ $(x; y)$ thỏa mãn phương trình bậc nhất hai ẩn $ax + by = c$ là một đường thẳng. Đường thẳng đó gọi là đường thẳng $ax + by = c$.
		\end{nhanxet}
	\end{tcth}
	
	\noindent\faBookmarkO~\textit{\textbf{Luện tập 2:}}
	
	\begin{tcth}
		\buocmot
		\item GV chia lớp thành ba nhóm lớn, mỗi nhóm chia thành nhóm nhỏ 3 – 4 HS ngồi gần nhau. Nhóm lớn 1, 2 và 3 lần lượt làm các ý a), b) và c).
		\buochai
		\item GV mời đại diện mỗi nhóm lên trình bày các ý a), b), c).
		\buocba
		\item Các nhóm khác nhận xét.
		\buocbon
		\item GV phân tích, nhận xét bài làm của HS. 
	\end{tcth}
