	\setlist[itemize]{%
		leftmargin=*, 
		topsep=3pt,
		itemsep=0pt
	}%
	\setlist[itemize,1]{label=\textbullet} % Tránh nhầm lẫn với dấu trừ.
	\setlist[itemize,2]{label=\faAngleRight}

		leftmargin=*, 
		topsep=3pt,
		itemsep=0pt
	}%
	\setlist[cacbuoc]{%
		label=\textit{\faCaretRight~Bước \arabic*.},
		wide=0pt,
		leftmargin=0pt, 
		labelsep=0.5em,
		itemsep=0pt, 
		topsep=3pt,
		before=\leavevmode
	}%
}
