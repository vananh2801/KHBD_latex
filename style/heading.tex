% Biến đếm (không dùng trực tiếp biến có sẵn sẽ gây lỗi trùng lặp với \section*{})
\makeatletter
% Phần
\newcounter{sb@part}
\pretocmd{\part}{\refstepcounter{sb@part}}{}{} % giá trị nhỏ nhất là 1
% Chương
\newcounter{sb@chapter}
\counterwithin*{sb@chapter}{sb@part}
\pretocmd{\chapter}{\refstepcounter{sb@chapter}}{}{} % giá trị nhỏ nhất là 1
% Bài dạy
\newcounter{sb@section}
\counterwithin*{sb@section}{sb@chapter}
\pretocmd{\section}{\refstepcounter{sb@section}}{}{} % giá trị nhỏ nhất là 1

% Lưu số lượng #1 trong section
% Arg: counter
\newcommand{\saveCounter}[1]{%
	% Nếu không có thì tạo counter mới
	\@ifundefined{c@#1@insection}{%
		\newcounter{#1@insection}
		\counterwithin*{#1@insection}{section}
	}{}%
	% Nếu có counter rồi thì tăng 1 đơn vị (giá trị nhỏ nhất là 1)
	\refstepcounter{#1@insection}%
	% Xuất lệnh \@importCounter ra .aux
	\immediate\write\@auxout{%
		\string\@importCounter%
		{#1}%
		{\number\value{sb@part}}%
		{\number\value{sb@chapter}}%
		{\number\value{sb@section}}%
		{\number\value{#1@insection}}%
	}%
}
% Lệnh này để gán bộ đếm của lần biên dịch trước bằng cách chạy lệnh từ .aux
% Arg: tên môi trường / sb@part / sb@chapter / sb@section / counter
\newcommand{\@importCounter}[5]{% 
	\expandafter\xdef\csname counter@#1@#2@#3@#4\endcsname{#5}%
}

% Lưu tiêu đề của môi trường
\newcommand{\saveTitle}[2]{%
	\immediate\write\@auxout{%
		\string\@importTitle%
		{#1}%
		{\number\value{sb@part}}%
		{\number\value{sb@chapter}}%
		{\number\value{sb@section}}%
		{\number\value{#1@insection}}%
		{#2}%
	}%
}
% Lệnh này để gán tiêu đề của lần biên dịch trước bằng cách chạy lệnh từ .aux
% Arg: tên môi trường / sb@part / sb@chapter / sb@section / nội dung tiêu đề
\newcommand{\@importTitle}[6]{%
	\expandafter\xdef\csname title@#1@#2@#3@#4@#5\endcsname{#6}%
}

% Lệnh lấy số lần dùng môi trường trong section hiện tại
% Trả về giá trị lớn nhất của #1@insection ở \saveCounter
\newcommand{\@getCounter}[1]{%
	\ifcsname counter@#1@\number\value{sb@part}@\number\value{sb@chapter}@\number\value{sb@section}\endcsname
		\csname counter@#1@\number\value{sb@part}@\number\value{sb@chapter}@\number\value{sb@section}\endcsname%
	\else
		0%
	\fi
}

% Danh sách tiết
\newcommand{\danhsachtiet}{%
	\@tempcnta=0\relax
	\ifthenelse{\equal{\@getCounter{subsubsection}}{0}}{%
		\textcolor{red}{\Large Thầy cô vui lòng soạn các tiết dạy ở phía sau hoặc biên dịch PDF 1 lần nữa để cập nhật danh sách tiết!}
	}{%
		Bài học này dạy trong \@getCounter{subsubsection}~tiết:
		\begin{itemize}
			\@whilenum\@tempcnta<\@getCounter{subsubsection}\relax\do{%
				\advance\@tempcnta by 1\relax
				\edef\@keytiet{title@subsubsection@\number\value{sb@part}@\number\value{sb@chapter}@\number\value{sb@section}@\the\@tempcnta}%
				\item Tiết \the\@tempcnta: \@themDauCham{\csname\@keytiet\endcsname}
			}%
		\end{itemize}
	}
}
\makeatother
