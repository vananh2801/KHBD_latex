% Biến đếm (không dùng trực tiếp biến có sẵn sẽ gây lỗi trùng lặp với \section*{})

\newcounter{sb@section}
\pretocmd{\part}{\refstepcounter{sb@part}}{}{}
\pretocmd{\chapter}{\refstepcounter{sb@chapter}}{}{}
\pretocmd{\section}{%
	\refstepcounter{sb@section}%
}{}{}
% Lưu số lượng
% Arg: counter
\newcommand{\saveCounter}[1]{%
	\immediate\write\@auxout{%
		\string\@importCounter%
		{#1}%
		{\number\value{sb@part}}%
		{\number\value{sb@chapter}}%
		{\number\value{sb@section}}%
		{\number\value{#1}}%
	}%
}
% Arg: sb@part / sb@chapter / sb@section / counter
\newcommand{\@importCounter}[5]{% 
	\expandafter\xdef\csname counter@#1@#2@#3@#4\endcsname{#5}%
}

% Lưu tiêu đề tiết
\newcommand{\saveTitle}[2]{%
	\immediate\write\@auxout{%
		\string\@importTitle%
		{#1}%
		{\number\value{sb@part}}%
		{\number\value{sb@chapter}}%
		{\number\value{sb@section}}%
		{\number\value{subsubsection}}%
		{#2}%
	}%
}
\newcommand{\@importTitle}[6]{%
	\expandafter\xdef\csname title@#1@#2@#3@#4@#5\endcsname{#6}%
}

% Lệnh lấy số tiết trong section hiện tại
\newcommand{\@printCounter}[1]{%
	\ifcsname counter@#1@\number\value{sb@part}@\number\value{sb@chapter}@\number\value{sb@section}\endcsname
	\csname counter@#1@\number\value{sb@part}@\number\value{sb@chapter}@\number\value{sb@section}\endcsname
	\else
	0%
	\fi
}
% Lưu nốt section cuối cùng
\AtEndDocument{\saveCounter{subsubsection}} 
\newcommand{\tientrinhdayhoc}{%
	Bài học này dạy trong \@printCounter{subsubsection}~tiết:
	\@tempcnta=0\relax
	\begin{itemize}
		\@whilenum\@tempcnta<\@printCounter{subsubsection}\relax\do{%
			\advance\@tempcnta by 1\relax
			\edef\@keytiet{title@subsubsection@\number\value{sb@part}@\number\value{sb@chapter}@\number\value{sb@section}@\the\@tempcnta}%
			\item Tiết \the\@tempcnta: \xoaxuongdong{\csname\@keytiet\endcsname}.
		}%
	\end{itemize}
}
% Xoá \\ trong nội dung
\newcommand{\xoaxuongdong}[1]{%
	\begingroup
	\let\\=\space      % Biến \\ thành dấu cách
	\let\newline=\space % Biến \newline thành dấu cách (nếu cần)
	#1%                % In tham số ra
	\endgroup
}
